Circuit Wiring
Connect Soil Sensor → NodeMCU ESP8266
Sensor VCC
↓
3.3V (NodeMCU)
Sensor GND
↓
GND (NodeMCU)
Sensor AOUT
↓
A0 (NodeMCU)
⚠️ NodeMCU's A0 pin only accepts 0–1V. Most sensors output 0–3.3V — if your sensor outputs 3.3V, use a voltage divider (10kΩ + 33kΩ resistors) between AOUT and A0.
📐 Voltage Divider (if needed)
Sensor AOUT ──── 10kΩ ──── A0 (NodeMCU)
                    │
                   33kΩ
                    │
                   GND

Output = 3.3V × (33kΩ / (10kΩ + 33kΩ)) = 3.3V × 0.77 = ~2.5V
→ Actually still above 1V limit! Use 33kΩ + 100kΩ for ~2.5V sensor.

Safest option: Use sensors specifically rated for NodeMCU (3.3V with A0-safe output).

NodeMCU Firmware Code
nodemcu/soil_monitor.ino — complete code
c++ — nodemcu/soil_monitor.ino 
/*
 * SoilSense IoT — NodeMCU ESP8266 Firmware
 * Reads soil moisture sensor → sends to Firebase Realtime Database
 * with timestamp every 5 seconds.
 *
 * Libraries: FirebaseESP8266, NTPClient, ArduinoJson
 */

#include <ESP8266WiFi.h>
#include <FirebaseESP8266.h>
#include <NTPClient.h>
#include <WiFiUdp.h>
#include <ArduinoJson.h>

// ── CONFIGURATION — Edit these values ──────────────────────────────
#define WIFI_SSID      "YOUR_WIFI_SSID"
#define WIFI_PASSWORD  "YOUR_WIFI_PASSWORD"

#define FIREBASE_HOST  "soilsense-iot-default-rtdb.firebaseio.com"  // without https://
#define FIREBASE_AUTH  "YOUR_DATABASE_SECRET_KEY"                     // from Project Settings

#define SENSOR_PIN     A0        // Analog pin (only A0 on NodeMCU)
#define SEND_INTERVAL  5000      // ms between readings (5 seconds)
#define IST_OFFSET     19800    // UTC+5:30 for India (seconds)
// ──────────────────────────────────────────────────────────────────

FirebaseData   fbData;
FirebaseConfig fbConfig;
FirebaseAuth   fbAuth;

WiFiUDP   ntpUDP;
NTPClient timeClient(ntpUDP, "pool.ntp.org", IST_OFFSET, 60000);

unsigned long lastSendTime = 0;
int           readingCount = 0;

void connectWiFi() {
  Serial.print("Connecting to WiFi");
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500); Serial.print(".");
  }
  Serial.println("\n✅ WiFi Connected!");
  Serial.print("IP Address: ");
  Serial.println(WiFi.localIP());
}

String getISOTimestamp() {
  timeClient.update();
  time_t epochTime = timeClient.getEpochTime();
  struct tm *ptm = gmtime(&epochTime);

  char buf[25];
  snprintf(buf, sizeof(buf),
    "%04d-%02d-%02dT%02d:%02d:%02d",
    ptm->tm_year + 1900,
    ptm->tm_mon  + 1,
    ptm->tm_mday,
    ptm->tm_hour,
    ptm->tm_min,
    ptm->tm_sec
  );
  return String(buf);
}

float readMoisture() {
  // Average 5 readings to reduce noise
  int total = 0;
  for (int i = 0; i < 5; i++) {
    total += analogRead(SENSOR_PIN);
    delay(10);
  }
  int raw = total / 5;

  // Map: dry soil = ~800 ADC = 0%, wet soil = ~300 ADC = 100%
  // Adjust these calibration values for YOUR sensor!
  const int DRY_VAL = 800;
  const int WET_VAL = 300;
  float moisture = map(raw, DRY_VAL, WET_VAL, 0, 100);
  return constrain(moisture, 0.0, 100.0);
}

void setup() {
  Serial.begin(115200);
  Serial.println("\n\n🌱 SoilSense IoT Starting...");

  connectWiFi();

  // Firebase config
  fbConfig.host = FIREBASE_HOST;
  fbConfig.signer.tokens.legacy_token = FIREBASE_AUTH;
  Firebase.begin(&fbConfig, &fbAuth);
  Firebase.reconnectWiFi(true);
  fbData.setResponseSize(4096);

  // NTP time
  timeClient.begin();
  timeClient.update();
  Serial.println("✅ Firebase + NTP Ready!");
}

void loop() {
  if (millis() - lastSendTime >= SEND_INTERVAL) {
    lastSendTime = millis();

    float  moisture  = readMoisture();
    int    raw       = analogRead(SENSOR_PIN);
    String timestamp = getISOTimestamp();
    long   epoch     = timeClient.getEpochTime();

    // Path: /sensor/moisture/{epochTimestamp}
    String path = "/sensor/moisture/" + String(epoch);

    // Write as JSON object using FirebaseJson
    FirebaseJson jsonData;
    jsonData.set("value",     moisture);
    jsonData.set("raw",       raw);
    jsonData.set("timestamp", timestamp);

    if (Firebase.setJSON(fbData, path, jsonData)) {
      readingCount++;
      Serial.printf("✅ [%d] Sent → %.1f%% (raw:%d) @ %s\n",
        readingCount, moisture, raw, timestamp.c_str());
    } else {
      Serial.print("❌ Firebase error: ");
      Serial.println(fbData.errorReason());
    }
  }
}
🎯 Calibration tip: To find your sensor's DRY_VAL and WET_VAL, open Serial Monitor and note the raw ADC reading when the sensor is in dry air and when fully submerged in water. Replace those constants in the code.

dashboard/app.js — Firebase connection

javascript — dashboard/app.js
// dashboard/app.js
// Connects to Firebase and drives the dashboard UI

// ── Init Firebase ──────────────────────────────────────────────────
const app = firebase.initializeApp(firebaseConfig);  // from firebase-config.js
const db  = firebase.database();

// ── Real-Time Listener ─────────────────────────────────────────────
// Fires every time a NEW reading is added by NodeMCU
db.ref(DB_PATH).limitToLast(1).on('child_added', (snapshot) => {
  const data = snapshot.val();
  if (data && data.value !== undefined) {
    updateDashboard(data.value, data.raw, data.timestamp);
  }
});

// ── Load Last 20 Readings on Page Load ────────────────────────────
db.ref(DB_PATH).limitToLast(20).once('value', (snapshot) => {
  const entries = [];
  snapshot.forEach(child => entries.push(child.val()));

  // Oldest → newest order for chart
  entries.reverse().forEach(d => {
    if (d && d.value !== undefined)
      pushToChart(d.value, d.timestamp);
  });
  realtimeChart.update();
  console.log(`✅ Loaded ${entries.length} historical readings`);
});

// ── updateDashboard(value, raw, timestamp) ─────────────────────────
function updateDashboard(value, raw, timestamp) {
  const ts      = timestamp ? new Date(timestamp) : new Date();
  const timeStr = ts.toLocaleTimeString('en-IN', { hour12: false });
  const pct     = parseFloat(value).toFixed(1);

  // Update DOM elements
  document.getElementById('moistureValue').textContent = pct;
  document.getElementById('rawValue').textContent      = raw || '--';
  document.getElementById('lastTime').textContent       = timeStr;

  // Status badge
  const status = getStatus(value);
  document.getElementById('statusBadge').textContent   = status.label;
  document.getElementById('statusBadge').className    = `badge badge-${status.cls}`;

  // Push to real-time chart
  pushToChart(value, timestamp);
  realtimeChart.update();

  // Alert if dry
  if (value < 30) triggerAlert(value);
}

function getStatus(v) {
  if (v < 20) return { label: '🔴 Critical Dry',   cls: 'critical' };
  if (v < 40) return { label: '🟡 Dry',           cls: 'dry'      };
  if (v < 70) return { label: '🟢 Optimal',      cls: 'ok'       };
  return           { label: '🔵 Wet',           cls: 'wet'      };
}

Troubleshooting
Common issues and fixes
ProblemCauseFix
Serial monitor garbage characters Wrong baud rate Set to 115200
NodeMCU not detected (COM port missing) Missing USB driver Install CP2102 or CH340 driver for your OS
❌ Firebase error: connection refused Wrong FIREBASE_HOST Remove https:// prefix from host
Sensor always reads 0% or 100% Wrong calibration values Read raw ADC in dry air and in water, set DRY_VAL and WET_VAL
Dashboard shows no data CORS or wrong DB path Open browser DevTools → Console for error; check DB_PATH matches Firebase
NodeMCU keeps restarting (crash loop) Insufficient power Use a good USB cable + 5V 1A adapter, not laptop USB
Firebase permission denied Database rules too strict Set rules to true for both read and write in test mode
