ESP32 Firmware Code
nodemcu/soil_monitor.ino — complete code
c++ — esp32/soil_monitor.ino 
/*
 * SoilSense Pro — ESP32 Firmware v2.3
 * Reads soil moisture + DHT11 → Firebase RTDB every 5s
 * Hardware: ESP32 Dev + Capacitive Soil Sensor + DHT11 + LED + Buzzer
 *
 * Libraries: FirebaseESP32 (Mobizt), DHT (Adafruit), NTPClient
 *
 * ESP32 Pin Map:
 *   GPIO34 — Soil Sensor AOUT
 *   GPIO4  — DHT11 DATA
 *   GPIO26 — LED (via 220Ω resistor)
 *   GPIO15 — Active Buzzer
 */

#include <WiFi.h>
#include <FirebaseESP32.h>
#include <DHT.h>
#include <NTPClient.h>
#include <WiFiUdp.h>

// ── CONFIGURATION — Edit these values ──────────────────────────────
#define WIFI_SSID      "YOUR_WIFI_SSID"
#define WIFI_PASSWORD  "YOUR_WIFI_PASSWORD"

#define FIREBASE_HOST "soilsense-iot-d1c31-default-rtdb.firebaseio.com"
#define FIREBASE_AUTH "YOUR_DATABASE_SECRET"

#define SOIL_ADC_PIN  34   // GPIO34 — Soil sensor AOUT
#define DHT_PIN       4    // GPIO4  — DHT11 DATA
#define LED_PIN       26   // GPIO26 — LED (via 220Ω)
#define BUZZER_PIN    15   // GPIO15 — Active Buzzer

// ── Calibration (adjust for your sensor) ────────────────────────
#define DRY_VALUE     4095  // ADC when sensor is in dry air
#define WET_VALUE     1800  // ADC when sensor is in water
#define DRY_THRESHOLD 30    // % moisture below which LED+buzzer alert
#define SEND_INTERVAL  5000  // ms between readings
#define IST_OFFSET     19800 // UTC+5:30 for India (seconds)

FirebaseData   fbData;
FirebaseConfig fbConfig;
FirebaseAuth   fbAuth;

DHT            dht(DHT_PIN, DHT11);
WiFiUDP        ntpUDP;
NTPClient timeClient(ntpUDP, "pool.ntp.org", IST_OFFSET, 60000);

unsigned long lastSendTime = 0;
int           readingCount = 0;

void connectWiFi() {
  Serial.print("Connecting to WiFi");
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500); Serial.print(".");
  }
  Serial.println("\n[WiFi] Connected — IP: " + WiFi.localIP().toString());
}

float computeHeatIndex(float t, float h) {
  float hi = t * 1.8 + 32;
  hi += h * 0.05;
  return (hi - 32) * 0.555;
}

void sendToFirebase(float moisture, int rawADC,
                       float temp, float hum,
                       float heatIdx, bool alert,
                       String timestamp, long epoch) {
  String path = "/sensor/moisture/" + String(epoch);
  FirebaseJson jsonData;
  jsonData.set("value",       moisture);
  jsonData.set("raw",         rawADC);
  jsonData.set("temperature", temp);
  jsonData.set("humidity",    hum);
  jsonData.set("heat_index",  heatIdx);
  jsonData.set("alert",       alert);
  jsonData.set("timestamp",  timestamp);

  if (Firebase.setJSON(fbData, path, jsonData)) {
    Serial.printf("[%d] OK | Soil:%.1f%% | Temp:%.1fC | Hum:%.1f%%\n",
                  readingCount, moisture, temp, hum);
  } else {
    Serial.print("[Firebase] Error: "); Serial.println(fbData.errorReason());
  }
}

void blinkLED(int times, int delayMs) {
  for (int i = 0; i < times; i++) {
    digitalWrite(LED_PIN, HIGH);
    if (times == 1) digitalWrite(BUZZER_PIN, HIGH);
    delay(delayMs);
    digitalWrite(LED_PIN, LOW);
    digitalWrite(BUZZER_PIN, LOW);
    delay(delayMs);
  }
}

void setup() {
  Serial.begin(115200);
  Serial.println("\n🌱 SoilSense Pro — ESP32 Firmware v2.3");

  pinMode(SOIL_ADC_PIN, INPUT);
  pinMode(LED_PIN, OUTPUT);
  pinMode(BUZZER_PIN, OUTPUT);
  digitalWrite(LED_PIN, LOW);
  digitalWrite(BUZZER_PIN, LOW);

  blinkLED(3, 200);
  blinkLED(2, 100);

  dht.begin();
  Serial.println("[DHT] DHT11 initialized on GPIO4");

  connectWiFi();

  fbConfig.host = FIREBASE_HOST;
  fbConfig.signer.tokens.legacy_token = FIREBASE_AUTH;
  Firebase.begin(&fbConfig, &fbAuth);
  Firebase.reconnectWiFi(true);

  timeClient.begin();
  timeClient.update();

  Serial.println("[Ready] Sending data every 5s...");
}

void loop() {
  if (millis() - lastSendTime >= SEND_INTERVAL) {
    lastSendTime = millis();
    readingCount++;

    // ── Soil Moisture ──
    int rawADC = analogRead(SOIL_ADC_PIN);
    float moisture = map(rawADC, DRY_VALUE, WET_VALUE, 0, 100);
    moisture = constrain(moisture, 0.0, 100.0);
    bool dryAlert = (moisture < DRY_THRESHOLD);

    // ── DHT11 ──
    float temp = dht.readTemperature();
    float hum  = dht.readHumidity();
    float heatIdx = computeHeatIndex((isnan(temp) ? 0 : temp),
                                          (isnan(hum)  ? 0 : hum));
    if (isnan(temp)) { temp = 0; hum = 0; }

    // ── LED + Buzzer alerts ──
    if (dryAlert) {
      blinkLED(3, 150);
      digitalWrite(LED_PIN, HIGH);
      blinkLED(3, 100);
      digitalWrite(LED_PIN, LOW);
    } else {
      digitalWrite(LED_PIN, LOW);
      digitalWrite(BUZZER_PIN, LOW);
    }

    // ── Timestamp ──
    timeClient.update();
    long   epoch    = timeClient.getEpochTime();
    time_t   rawTime = epoch;
    struct tm *ptm  = localtime(&rawTime);
    char   buf[32];
    snprintf(buf, sizeof(buf), "%04d-%02d-%02dT%02d:%02d:%02d",
             ptm->tm_year + 1900, ptm->tm_mon + 1, ptm->tm_mday,
             ptm->tm_hour, ptm->tm_min, ptm->tm_sec);
    String timestamp = String(buf);

    sendToFirebase(moisture, rawADC, temp, hum, heatIdx, dryAlert, timestamp, epoch);
  }
}
🎯 Calibration tip: To find your sensor's DRY_VAL and WET_VAL, open Serial Monitor and note the raw ADC reading when the sensor is in dry air and when fully submerged in water. Replace those constants in the code.

Testing & Running
How to verify everything works
🔌

Step 1 — Upload Firmware

Open soil_monitor.ino in Arduino IDE → Fill in WiFi + Firebase credentials → Connect NodeMCU via USB → Select correct COM port → Click Upload (→)

📟

Step 2 — Check Serial Monitor

Tools → Serial Monitor → Set baud rate to 115200. You should see:
✅ WiFi Connected! then ✅ [1] Sent → 72.4% (raw:285) @ 2024-01-13T10:30:56

🔥

Step 3 — Verify Firebase

Firebase Console → Realtime Database → Data tab. You should see /sensor/moisture/{timestamp} nodes appearing every 5 seconds.

🌐

Step 4 — Open Dashboard

Double-click dashboard/index.html to open in browser (or use Live Server in VS Code). The gauge and chart should update in real time!

🚀 To deploy the dashboard online for free: upload dashboard/ folder to GitHub Pages or Firebase Hosting (firebase deploy). Your dashboard becomes accessible from anywhere!

Troubleshooting
Common issues and fixes
ProblemCauseFix
Serial monitor garbage characters Wrong baud rate Set to 115200
ESP32 not detected (COM port missing) Missing USB driver Install CP2102 or CH340 driver for your OS
❌ Firebase error: connection refused Wrong FIREBASE_HOST Remove https:// prefix from host
Sensor always reads 0% or 100% Wrong calibration values Read raw ADC in dry air and in water, set DRY_VAL and WET_VAL
Dashboard shows no data CORS or wrong DB path Open browser DevTools → Console for error; check DB_PATH matches Firebase
ESP32 keeps restarting (crash loop) Insufficient power Use a good USB cable + 5V 1A adapter, not laptop USB
Firebase permission denied Database rules too strict Set rules to true for both read and write in test mode
